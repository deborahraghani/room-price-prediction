import numpy as np
import pandas as pd
from sklearn.model_selection import train_test_split, KFold, cross_val_score
from sklearn.linear_model import LinearRegression
from sklearn.metrics import mean_squared_error, r2_score

# Assuming mldata_no_outliers is already defined and preprocessed
X = mldata_no_outliers[['person_capacity', 'multi', 'biz', 'cleanliness_rating', 
                         'guest_satisfaction_overall', 'bedrooms', 'metro_dist', 
                         'log_tranformation', 'room_type_Entire home/apt', 
                         'room_type_Private room', 'room_type_Shared room', 
                         'room_shared_False', 'room_shared_True', 
                         'host_is_superhost_False', 'host_is_superhost_True', 
                         'room_private_False', 'room_private_True']]
y = mldata_no_outliers['log_tranformation of realsum']

# Initialize the linear regression model
lr = LinearRegression()

# Set up K-Folds
k = 5  # You can change this value based on your needs
kf = KFold(n_splits=k, shuffle=True, random_state=42)

# List to store RMSE and R-squared for each fold
rmse_scores = []
r2_scores = []

# Perform cross-validation
for train_index, test_index in kf.split(X):
    X_train, X_test = X.iloc[train_index], X.iloc[test_index]
    y_train, y_test = y.iloc[train_index], y.iloc[test_index]

    # Fit the model to the training data
    lr.fit(X_train, y_train)
    
    # Make predictions on the test set
    y_test_pred = lr.predict(X_test)
    
    # Calculate RMSE for the test set
    fold_rmse = np.sqrt(mean_squared_error(y_test, y_test_pred))
    rmse_scores.append(fold_rmse)
    
    # Calculate R-squared for the test set
    fold_r2 = r2_score(y_test, y_test_pred)
    r2_scores.append(fold_r2)

# Print the results
print("K-Fold RMSE Scores:", rmse_scores)
print("Average RMSE:", np.mean(rmse_scores))
print("K-Fold R-squared Scores:", r2_scores)
print("Average R-squared:", np.mean(r2_scores))
